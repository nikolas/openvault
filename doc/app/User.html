def work_string
  if self.title.blank? && self.organization.blank?
    ""
  elsif self.title.blank? && !self.organization.blank?
    "works at #{self.organization}"
  elsif !self.title.blank? && self.organization.blank?
    "#{self.title}"
  else
    "#{self.title} at #{self.organization}"
  end
end

def self.find id
  parsed_json = ActiveSupport::JSON.decode(open(self.endpoint_uri(id)).read)
  raise "Missing Wordpress Post #{ self.endpoint_uri(id)}" unless parsed_json["status"] == "ok"
  Post.new(parsed_json)
end
